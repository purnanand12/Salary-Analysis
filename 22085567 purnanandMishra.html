def calculate_numeric_correlation(data):
    # Select only columns with numeric types (integers and floats)
    numeric_columns = data.select_dtypes(include=['int', 'float']) 
    # Calculate correlation 
    correlation = numeric_columns.corr()
    return correlation

# Call the function
